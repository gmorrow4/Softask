function fmtScore(n){if(n===null||n===undefined)return'—';if(n===0)return'E';return n>0?'+'+n:''+n;}
function scoreClass(n){if(n===0)return'score-ev';return n<0?'score-neg':'score-pos';}
function moveTxt(m){if(m===0)return'<span class="move-ev">—</span>';if(m>0)return'<span class="move-up">▲'+m+'</span>';return'<span class="move-dn">▼'+Math.abs(m)+'</span>';}
function medalEl(pos){
  if(pos===1)return'<span class="medal medal-g">1</span>';
  if(pos===2)return'<span class="medal medal-s">2</span>';
  if(pos===3)return'<span class="medal medal-b">3</span>';
  return'<span style="font-size:13px;font-weight:500;color:#888">'+pos+'</span>';
}
function teeEl(t){
  if(t==='gold')return'<span class="tee-g">G</span>';
  return'<span class="tee-s">S</span>';
}

function renderRows(data){
  const body=document.getElementById('tbl-body');
  body.innerHTML='';
  let lastSec='';
  data.forEach(t=>{
    const total=t.d1+(t.d2||0);
    let sec='';
    if(t.pos===1)sec='Podium contenders';
    else if(total<0&&lastSec!=='Under par')sec='Under par';
    else if(total===0&&lastSec!=='Even par')sec='Even par';
    else if(total>0&&lastSec!=='Over par')sec='Over par';
    if(sec&&sec!==lastSec){
      lastSec=sec;
      const dr=document.createElement('tr');
      dr.innerHTML='<td colspan="6" class="section-divider">'+sec+'</td>';
      body.appendChild(dr);
    }
    const tr=document.createElement('tr');
    if(t.pos<=3)tr.className='top3';
    tr.dataset.tee=t.tee;
    tr.dataset.score=total;
    const d2txt='<span class="'+scoreClass(t.d2)+'">'+fmtScore(t.d2)+'</span><br><small style="font-size:10px">'+moveTxt(t.move)+'</small>';
    tr.innerHTML=`
      <td style="text-align:center">${medalEl(t.pos)}</td>
      <td>
        <div class="team-col">
          <div class="av" style="background:${t.avc};color:${t.avt}">${t.av}</div>
          <div>
            <div class="team-names">${t.p1} / ${t.p2}</div>
            <div class="team-sub">HCP ${t.hcp} · ${teeEl(t.tee)}</div>
          </div>
        </div>
      </td>
      <td class="r"><span class="${scoreClass(t.d1)}" style="font-size:13px;font-weight:500">${fmtScore(t.d1)}</span></td>
      <td class="r" style="line-height:1.4">${d2txt}</td>
      <td class="r" style="font-size:14px;font-weight:500" class="${scoreClass(total)}"><span class="${scoreClass(total)}">${fmtScore(total)}</span></td>
      <td class="r"><span class="thru-cell">${t.thru}</span></td>
    `;
    body.appendChild(tr);
  });
}

function filterTbl(f,el){
  document.querySelectorAll('.fbtn').forEach(b=>b.classList.remove('active'));
  el.classList.add('active');
  let filtered=teams;
  if(f==='under')filtered=teams.filter(t=>(t.d1+(t.d2||0))<0);
  else if(f==='over')filtered=teams.filter(t=>(t.d1+(t.d2||0))>0);
  else if(f==='gold')filtered=teams.filter(t=>t.tee==='gold');
  else if(f==='silver')filtered=teams.filter(t=>t.tee==='silver');
  renderRows(filtered);
}
